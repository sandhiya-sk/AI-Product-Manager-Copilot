### 1.2 Create Table: `users`

```sql
CREATE TABLE IF NOT EXISTS users (
    -- -------------------------------------------------------
    -- PRIMARY KEY
    -- UUID auto-generated by pgcrypto extension
    -- -------------------------------------------------------
    user_id             UUID            PRIMARY KEY DEFAULT gen_random_uuid(),

    -- -------------------------------------------------------
    -- IDENTITY COLUMNS
    -- -------------------------------------------------------
    email               VARCHAR(255)    NOT NULL,           -- Login email; must be globally unique
    password_hash       VARCHAR(255)    NOT NULL,           -- bcrypt hash of password (60-char output)
    full_name           VARCHAR(255)    NULL,               -- Display name; optional

    -- -------------------------------------------------------
    -- ROLE-BASED ACCESS CONTROL
    -- Enforced via CHECK constraint; no separate roles table needed
    -- -------------------------------------------------------
    role                VARCHAR(50)     NOT NULL
                            CHECK (role IN ('product_manager', 'customer')),

    -- -------------------------------------------------------
    -- PROJECT ASSOCIATION
    -- Optional; can be assigned later via UPDATE
    -- -------------------------------------------------------
    project_id          UUID            NULL,

    -- -------------------------------------------------------
    -- ACCOUNT LIFECYCLE
    -- Soft delete: set is_active = FALSE instead of hard DELETE
    -- -------------------------------------------------------
    is_active           BOOLEAN         NOT NULL DEFAULT TRUE,

    -- -------------------------------------------------------
    -- AUDIT TIMESTAMPS (all UTC via TIMESTAMPTZ)
    -- -------------------------------------------------------
    created_at          TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    last_login_at       TIMESTAMPTZ     NULL,               -- Updated on every successful login

    -- -------------------------------------------------------
    -- UNIQUE CONSTRAINT on email (also covered by index below)
    -- -------------------------------------------------------
    CONSTRAINT uq_users_email UNIQUE (email)
);

-- B-Tree index on email (used for login lookups)
CREATE INDEX IF NOT EXISTS idx_users_email
    ON users (email);

-- B-Tree index on role (used for role-filtered queries)
CREATE INDEX IF NOT EXISTS idx_users_role
    ON users (role);

-- B-Tree index on project_id (used to list all users in a project)
CREATE INDEX IF NOT EXISTS idx_users_project_id
    ON users (project_id);

-- Table and column comments
COMMENT ON TABLE  users               IS 'Stores both product_manager and customer accounts with RBAC.';
COMMENT ON COLUMN users.user_id       IS 'UUID primary key, auto-generated by gen_random_uuid()';
COMMENT ON COLUMN users.email         IS 'Unique login email address; max 255 chars';
COMMENT ON COLUMN users.password_hash IS 'bcrypt hashed password; never store plaintext';
COMMENT ON COLUMN users.full_name     IS 'Display name; optional';
COMMENT ON COLUMN users.role          IS 'RBAC role: product_manager or customer';
COMMENT ON COLUMN users.project_id    IS 'Default project UUID this user belongs to; nullable';
COMMENT ON COLUMN users.is_active     IS 'Soft-delete flag: FALSE disables login without deleting the row';
COMMENT ON COLUMN users.created_at    IS 'UTC timestamp of account creation';
COMMENT ON COLUMN users.updated_at    IS 'UTC timestamp of last record modification (auto-updated by trigger)';
COMMENT ON COLUMN users.last_login_at IS 'UTC timestamp of the most recent successful login';
```

### 1.3 Create Table: `raw_feedback`

```sql
CREATE TABLE IF NOT EXISTS raw_feedback (
    -- -------------------------------------------------------
    -- PRIMARY KEY
    -- -------------------------------------------------------
    feedback_id             UUID            PRIMARY KEY DEFAULT gen_random_uuid(),

    -- -------------------------------------------------------
    -- SOURCE & SUBMITTER INFORMATION
    -- source: how the feedback arrived (CSV file or web form)
    -- submitted_by_role: role of the person who submitted
    -- -------------------------------------------------------
    source                  VARCHAR(50)     NOT NULL
                                CHECK (source IN ('csv_upload', 'text_form')),

    submitted_by_role       VARCHAR(50)     NOT NULL
                                CHECK (submitted_by_role IN ('product_manager', 'customer')),

    user_id                 UUID            NOT NULL
                                REFERENCES users(user_id)
                                ON DELETE SET NULL
                                DEFERRABLE INITIALLY IMMEDIATE,
    -- ON DELETE SET NULL: keeps feedback even if the user account is deleted

    project_id              UUID            NOT NULL,
    -- UUID identifying the product/project this feedback belongs to

    -- -------------------------------------------------------
    -- CUSTOMER INFORMATION
    -- Only relevant for CSV rows (where customer data is embedded)
    -- NULL for text_form submissions
    -- -------------------------------------------------------
    customer_name           VARCHAR(255)    NULL,
    customer_email          VARCHAR(255)    NULL,

    -- -------------------------------------------------------
    -- CORE FEEDBACK CONTENT
    -- Both are required; TEXT type = unlimited length
    -- -------------------------------------------------------
    subject                 TEXT            NOT NULL
                                CONSTRAINT chk_raw_subject_not_empty CHECK (LENGTH(TRIM(subject)) > 0),
    description             TEXT            NOT NULL
                                CONSTRAINT chk_raw_description_not_empty CHECK (LENGTH(TRIM(description)) > 0),

    -- -------------------------------------------------------
    -- STRUCTURED CLASSIFICATION FIELDS
    -- All normalized to defined enum values
    -- -------------------------------------------------------
    priority                VARCHAR(50)     NOT NULL DEFAULT 'Medium'
                                CHECK (priority IN ('Low', 'Medium', 'High', 'Critical')),

    category                VARCHAR(100)    NOT NULL DEFAULT 'General'
                                CHECK (category IN (
                                    'Bug',
                                    'Feature Request',
                                    'Improvement',
                                    'Complaint',
                                    'General'
                                )),

    product_name            VARCHAR(255)    NULL,
    product_version         VARCHAR(100)    NULL,
    submission_date         DATE            NULL,           -- Original feedback date from CSV; YYYY-MM-DD

    tags                    TEXT[]          NULL,
    -- PostgreSQL TEXT array; e.g. ARRAY['login','performance','mobile']
    -- Queried with @> (contains) or unnest() operators

    sentiment_self_reported VARCHAR(50)     NULL
                                CHECK (
                                    sentiment_self_reported IN ('Positive', 'Negative', 'Neutral')
                                    OR sentiment_self_reported IS NULL
                                ),

    language                VARCHAR(10)     NOT NULL DEFAULT 'en',
    -- ISO 639-1 language code; default 'en' (English)

    -- -------------------------------------------------------
    -- INGESTION AUDIT METADATA
    -- -------------------------------------------------------
    upload_timestamp        TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    -- When the record was inserted (auto-set by DB)

    raw_metadata            JSONB           NOT NULL DEFAULT '{}',
    -- Flexible JSONB blob. Expected keys:
    --   original_filename  TEXT     — CSV filename, null for text_form
    --   file_row_number    INTEGER  — Row index in the CSV file
    --   ip_address         TEXT     — Client IP from Flask request
    --   user_agent         TEXT     — Browser user-agent string
    --   batch_id           TEXT     — UUID shared by all rows from one CSV upload

    canonical_json          JSONB           NOT NULL DEFAULT '{}',
    -- Full canonical JSON snapshot of the record at ingestion time
    -- Enables record replay without relying on individual columns

    -- -------------------------------------------------------
    -- DEDUPLICATION FIELDS (set by Module 3 pipeline)
    -- -------------------------------------------------------
    weight                  INTEGER         NOT NULL DEFAULT 1
                                CHECK (weight >= 1),
    -- Semantic frequency counter.
    -- Starts at 1 (the record itself).
    -- Incremented by +1 for each semantically duplicate record merged into it.

    duplicate_group_id      UUID            NULL,
    -- NULL until Module 3 deduplication runs.
    -- After: shared UUID for all records in the same semantic cluster.

    -- -------------------------------------------------------
    -- PROCESSING STATUS LIFECYCLE
    -- Transitions: pending → processing → processed | duplicate | failed
    -- -------------------------------------------------------
    processing_status       VARCHAR(50)     NOT NULL DEFAULT 'pending'
                                CHECK (processing_status IN (
                                    'pending',      -- Awaiting Module 3
                                    'processing',   -- Currently in Module 3 pipeline
                                    'processed',    -- Pipeline completed; row exists in processed_feedback
                                    'duplicate',    -- Flagged as duplicate; weight added to canonical
                                    'failed'        -- Pipeline error; see processing_error column
                                )),

    processing_error        TEXT            NULL,
    -- Populated with exception message when processing_status = 'failed'

    processed_at            TIMESTAMPTZ     NULL,
    -- Set when processing_status changes to 'processed' or 'duplicate'

    -- -------------------------------------------------------
    -- AUDIT TIMESTAMPS
    -- -------------------------------------------------------
    created_at              TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    updated_at              TIMESTAMPTZ     NOT NULL DEFAULT NOW()
    -- updated_at is auto-maintained by trigger trg_raw_feedback_updated_at
);

-- -------------------------------------------------------
-- INDEXES for raw_feedback
-- -------------------------------------------------------

-- 1. Processing status — used heavily by Module 3 to fetch pending records
CREATE INDEX IF NOT EXISTS idx_raw_feedback_processing_status
    ON raw_feedback (processing_status);

-- 2. Project ID — filter all feedback for a given project
CREATE INDEX IF NOT EXISTS idx_raw_feedback_project_id
    ON raw_feedback (project_id);

-- 3. User ID — find all feedback submitted by a specific user
CREATE INDEX IF NOT EXISTS idx_raw_feedback_user_id
    ON raw_feedback (user_id);

-- 4. Upload timestamp ASC — Module 3 picks the oldest record as canonical
CREATE INDEX IF NOT EXISTS idx_raw_feedback_upload_timestamp
    ON raw_feedback (upload_timestamp ASC);

-- 5. Duplicate group ID (partial) — only on rows where grouping is assigned
CREATE INDEX IF NOT EXISTS idx_raw_feedback_duplicate_group_id
    ON raw_feedback (duplicate_group_id)
    WHERE duplicate_group_id IS NOT NULL;

-- 6. Source type — filter CSV vs. form submissions
CREATE INDEX IF NOT EXISTS idx_raw_feedback_source
    ON raw_feedback (source);

-- 7. Composite: category + priority — used in dashboard breakdowns
CREATE INDEX IF NOT EXISTS idx_raw_feedback_category_priority
    ON raw_feedback (category, priority);

-- 8. Composite: project + status — common combined filter
CREATE INDEX IF NOT EXISTS idx_raw_feedback_project_status
    ON raw_feedback (project_id, processing_status);

-- 9. JSONB GIN index on batch_id — query all rows from one CSV upload
CREATE INDEX IF NOT EXISTS idx_raw_feedback_batch_id
    ON raw_feedback USING GIN ((raw_metadata -> 'batch_id'));

-- 10. Full-text GIN on subject
CREATE INDEX IF NOT EXISTS idx_raw_feedback_subject_fts
    ON raw_feedback USING GIN (to_tsvector('english', subject));

-- 11. Full-text GIN on description
CREATE INDEX IF NOT EXISTS idx_raw_feedback_description_fts
    ON raw_feedback USING GIN (to_tsvector('english', description));

-- Comments
COMMENT ON TABLE  raw_feedback                       IS 'Raw ingested feedback. Source of truth. Records wait here until Module 3 processes them.';
COMMENT ON COLUMN raw_feedback.feedback_id           IS 'UUID primary key auto-generated per record';
COMMENT ON COLUMN raw_feedback.source                IS 'csv_upload (PM file) or text_form (web form submission)';
COMMENT ON COLUMN raw_feedback.submitted_by_role     IS 'product_manager or customer role of submitter';
COMMENT ON COLUMN raw_feedback.user_id               IS 'FK → users.user_id; NULL if user deleted';
COMMENT ON COLUMN raw_feedback.project_id            IS 'UUID of associated product/project';
COMMENT ON COLUMN raw_feedback.subject               IS 'Short feedback title; required; must be non-empty';
COMMENT ON COLUMN raw_feedback.description           IS 'Full feedback body; required; must be non-empty';
COMMENT ON COLUMN raw_feedback.priority              IS 'Low | Medium | High | Critical (default Medium)';
COMMENT ON COLUMN raw_feedback.category              IS 'Bug | Feature Request | Improvement | Complaint | General';
COMMENT ON COLUMN raw_feedback.tags                  IS 'PostgreSQL TEXT array of tag strings';
COMMENT ON COLUMN raw_feedback.language              IS 'ISO 639-1 code (default en)';
COMMENT ON COLUMN raw_feedback.upload_timestamp      IS 'UTC time of insert; used by Module 3 to select canonical record';
COMMENT ON COLUMN raw_feedback.raw_metadata          IS 'JSONB: filename, row_number, ip_address, user_agent, batch_id';
COMMENT ON COLUMN raw_feedback.canonical_json        IS 'Full normalized JSON snapshot at ingestion time';
COMMENT ON COLUMN raw_feedback.weight                IS 'Duplicate frequency counter; starts at 1; incremented by Module 3';
COMMENT ON COLUMN raw_feedback.duplicate_group_id    IS 'UUID of semantic group; NULL until Module 3 assigns it';
COMMENT ON COLUMN raw_feedback.processing_status     IS 'Lifecycle: pending → processing → processed|duplicate|failed';
COMMENT ON COLUMN raw_feedback.processing_error      IS 'Error message if processing_status = failed; else NULL';
COMMENT ON COLUMN raw_feedback.processed_at          IS 'UTC time when Module 3 finished this record';
```

### 1.4 Create Table: `processed_feedback`

```sql
CREATE TABLE IF NOT EXISTS processed_feedback (
    -- -------------------------------------------------------
    -- PRIMARY KEY
    -- -------------------------------------------------------
    processed_id                UUID            PRIMARY KEY DEFAULT gen_random_uuid(),

    -- -------------------------------------------------------
    -- FOREIGN KEY LINK TO RAW FEEDBACK
    -- CASCADE DELETE: if raw record is deleted, processed record is too
    -- -------------------------------------------------------
    raw_feedback_id             UUID            NOT NULL
                                    REFERENCES raw_feedback(feedback_id)
                                    ON DELETE CASCADE
                                    DEFERRABLE INITIALLY IMMEDIATE,

    -- -------------------------------------------------------
    -- SOURCE & IDENTITY (inherited from raw_feedback)
    -- -------------------------------------------------------
    source                      VARCHAR(50)     NOT NULL
                                    CHECK (source IN ('csv_upload', 'text_form')),

    submitted_by_role           VARCHAR(50)     NOT NULL
                                    CHECK (submitted_by_role IN ('product_manager', 'customer')),

    user_id                     UUID            NOT NULL,
    project_id                  UUID            NOT NULL,

    -- -------------------------------------------------------
    -- ORIGINAL TEXT (preserved verbatim from raw_feedback)
    -- -------------------------------------------------------
    original_subject            TEXT            NOT NULL,
    original_description        TEXT            NOT NULL,

    -- -------------------------------------------------------
    -- NLP PIPELINE TEXT STAGES
    -- -------------------------------------------------------
    clean_text                  TEXT            NOT NULL,
    -- Stage 1: HTML, URL, emoji, extra-space removal applied to original_description

    standardized_text           TEXT            NOT NULL,
    -- Stage 2: lowercase + date normalization applied to clean_text

    -- -------------------------------------------------------
    -- NLP OUTPUT ARRAYS (PostgreSQL TEXT[])
    -- -------------------------------------------------------
    tokens                      TEXT[]          NOT NULL DEFAULT '{}',
    -- NLTK word_tokenize output; stop words removed; only alpha tokens
    -- Example: ARRAY['login','crash','mobile','application']

    lemmas                      TEXT[]          NOT NULL DEFAULT '{}',
    -- spaCy en_core_web_sm lemmatization output
    -- Example: ARRAY['login','crash','mobil','applic']

    -- -------------------------------------------------------
    -- STRUCTURED METADATA (normalized, inherited & validated)
    -- -------------------------------------------------------
    priority                    VARCHAR(50)     NOT NULL DEFAULT 'Medium'
                                    CHECK (priority IN ('Low', 'Medium', 'High', 'Critical')),

    category                    VARCHAR(100)    NOT NULL DEFAULT 'General'
                                    CHECK (category IN (
                                        'Bug',
                                        'Feature Request',
                                        'Improvement',
                                        'Complaint',
                                        'General'
                                    )),

    product_name                VARCHAR(255)    NULL,
    product_version             VARCHAR(100)    NULL,
    tags                        TEXT[]          NULL,
    sentiment_self_reported     VARCHAR(50)     NULL
                                    CHECK (
                                        sentiment_self_reported IN ('Positive', 'Negative', 'Neutral')
                                        OR sentiment_self_reported IS NULL
                                    ),
    language                    VARCHAR(10)     NOT NULL DEFAULT 'en',
    submission_date             DATE            NULL,

    -- -------------------------------------------------------
    -- DEDUPLICATION FIELDS (from Module 3 deduplication engine)
    -- -------------------------------------------------------
    weight                      INTEGER         NOT NULL DEFAULT 1
                                    CHECK (weight >= 1),
    -- Final aggregated weight = number of semantically similar records collapsed into this one

    duplicate_group_id          UUID            NOT NULL,
    -- Always populated in processed_feedback (canonical always has a group UUID)

    -- -------------------------------------------------------
    -- TEXT STATISTICS (computed by Module 3 pipeline)
    -- -------------------------------------------------------
    word_count                  INTEGER         NOT NULL DEFAULT 0
                                    CHECK (word_count >= 0),
    -- len(clean_text.split())

    char_count                  INTEGER         NOT NULL DEFAULT 0
                                    CHECK (char_count >= 0),
    -- len(clean_text)

    token_count                 INTEGER         NOT NULL DEFAULT 0
                                    CHECK (token_count >= 0),
    -- len(tokens)

    lemma_count                 INTEGER         NOT NULL DEFAULT 0
                                    CHECK (lemma_count >= 0),
    -- len(lemmas)

    -- -------------------------------------------------------
    -- PROCESSING METADATA (JSONB)
    -- -------------------------------------------------------
    processing_metadata         JSONB           NOT NULL DEFAULT '{}',
    -- Expected keys:
    --   nlp_model_version        TEXT     — 'spacy-en_core_web_sm-3'
    --   embedding_model          TEXT     — 'all-MiniLM-L6-v2'
    --   similarity_threshold_used FLOAT   — e.g., 0.85
    --   pipeline_version         TEXT     — e.g., '1.0.0'
    --   processing_duration_ms   INTEGER  — elapsed time in ms

    processing_status           VARCHAR(50)     NOT NULL DEFAULT 'processed'
                                    CHECK (processing_status IN (
                                        'processed',    -- Normal completion
                                        'failed',       -- Pipeline error
                                        'reprocessing'  -- Being reprocessed (future use)
                                    )),

    processing_error            TEXT            NULL,
    -- Populated only when processing_status = 'failed'

    -- -------------------------------------------------------
    -- FULL-TEXT SEARCH VECTOR (auto-maintained by trigger)
    -- -------------------------------------------------------
    search_vector               TSVECTOR        NULL,
    -- Weighted: original_subject gets weight 'A', clean_text gets weight 'B'
    -- Automatically updated by trigger trg_processed_feedback_search_vector

    -- -------------------------------------------------------
    -- TIMESTAMPS
    -- -------------------------------------------------------
    processing_timestamp        TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    -- When Module 3 pipeline inserted this record

    created_at                  TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    updated_at                  TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    -- updated_at auto-maintained by trigger trg_processed_feedback_updated_at

    -- -------------------------------------------------------
    -- MODULE 4 HANDOFF FLAGS
    -- -------------------------------------------------------
    ready_for_classification    BOOLEAN         NOT NULL DEFAULT TRUE,
    -- TRUE after Module 3 completes NLP processing
    -- Module 4 queries WHERE ready_for_classification = TRUE AND classified_at IS NULL

    classified_at               TIMESTAMPTZ     NULL
    -- Set by Module 4 upon completing classification + theme extraction
    -- NULL = not yet processed by Module 4
);

-- -------------------------------------------------------
-- INDEXES for processed_feedback
-- -------------------------------------------------------

-- 1. Module 4 readiness (partial index — only unclassified ready rows)
CREATE INDEX IF NOT EXISTS idx_processed_feedback_ready
    ON processed_feedback (ready_for_classification, classified_at)
    WHERE ready_for_classification = TRUE AND classified_at IS NULL;

-- 2. Project ID filter
CREATE INDEX IF NOT EXISTS idx_processed_feedback_project_id
    ON processed_feedback (project_id);

-- 3. Duplicate group ID — join all members of a semantic cluster
CREATE INDEX IF NOT EXISTS idx_processed_feedback_dup_group
    ON processed_feedback (duplicate_group_id);

-- 4. Weight DESC — top-weighted (most-demanded) features first
CREATE INDEX IF NOT EXISTS idx_processed_feedback_weight
    ON processed_feedback (weight DESC);

-- 5. Category filter
CREATE INDEX IF NOT EXISTS idx_processed_feedback_category
    ON processed_feedback (category);

-- 6. Priority filter
CREATE INDEX IF NOT EXISTS idx_processed_feedback_priority
    ON processed_feedback (priority);

-- 7. FK back to raw_feedback
CREATE INDEX IF NOT EXISTS idx_processed_feedback_raw_id
    ON processed_feedback (raw_feedback_id);

-- 8. Full-text search on search_vector (GIN)
CREATE INDEX IF NOT EXISTS idx_processed_feedback_fts
    ON processed_feedback USING GIN (search_vector);

-- 9. Processing status filter
CREATE INDEX IF NOT EXISTS idx_processed_feedback_status
    ON processed_feedback (processing_status);

-- 10. Processing timestamp DESC — latest processed first
CREATE INDEX IF NOT EXISTS idx_processed_feedback_ts
    ON processed_feedback (processing_timestamp DESC);

-- 11. Composite: project + weight — top feedback per project
CREATE INDEX IF NOT EXISTS idx_processed_feedback_project_weight
    ON processed_feedback (project_id, weight DESC);

-- 12. Token array GIN — query by specific token membership
CREATE INDEX IF NOT EXISTS idx_processed_feedback_tokens_gin
    ON processed_feedback USING GIN (tokens);

-- 13. Lemma array GIN — query by specific lemma membership
CREATE INDEX IF NOT EXISTS idx_processed_feedback_lemmas_gin
    ON processed_feedback USING GIN (lemmas);

-- Comments
COMMENT ON TABLE  processed_feedback                          IS 'NLP-processed canonical feedback. One row per semantic cluster. Ready for Module 4.';
COMMENT ON COLUMN processed_feedback.processed_id            IS 'UUID primary key for this processed record';
COMMENT ON COLUMN processed_feedback.raw_feedback_id         IS 'FK → raw_feedback.feedback_id (canonical record)';
COMMENT ON COLUMN processed_feedback.original_subject        IS 'Verbatim subject from raw_feedback (unchanged)';
COMMENT ON COLUMN processed_feedback.original_description    IS 'Verbatim description from raw_feedback (unchanged)';
COMMENT ON COLUMN processed_feedback.clean_text              IS 'Description after HTML/URL/emoji/whitespace cleaning';
COMMENT ON COLUMN processed_feedback.standardized_text       IS 'clean_text after lowercase and normalization';
COMMENT ON COLUMN processed_feedback.tokens                  IS 'NLTK token list; stop words removed; TEXT array';
COMMENT ON COLUMN processed_feedback.lemmas                  IS 'spaCy lemmatized token list; TEXT array';
COMMENT ON COLUMN processed_feedback.weight                  IS 'Count of semantically duplicate records merged here';
COMMENT ON COLUMN processed_feedback.duplicate_group_id      IS 'UUID of semantic duplicate group; always set';
COMMENT ON COLUMN processed_feedback.word_count              IS 'Word count of clean_text';
COMMENT ON COLUMN processed_feedback.char_count              IS 'Character count of clean_text';
COMMENT ON COLUMN processed_feedback.token_count             IS 'Count of tokens array elements';
COMMENT ON COLUMN processed_feedback.lemma_count             IS 'Count of lemmas array elements';
COMMENT ON COLUMN processed_feedback.processing_metadata     IS 'JSONB: NLP model info, pipeline version, duration_ms';
COMMENT ON COLUMN processed_feedback.search_vector           IS 'Weighted tsvector; A=subject, B=clean_text; auto-updated by trigger';
COMMENT ON COLUMN processed_feedback.ready_for_classification IS 'TRUE when NLP complete; Module 4 reads this';
COMMENT ON COLUMN processed_feedback.classified_at           IS 'Set by Module 4; NULL = not yet classified';
```

### 1.5 Trigger Functions

```sql
-- -------------------------------------------------------
-- TRIGGER FUNCTION: auto-update search_vector
-- Fires BEFORE INSERT or UPDATE on original_subject or clean_text
-- Weight A = subject (higher rank), Weight B = clean_text (lower rank)
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION update_processed_feedback_search_vector()
RETURNS TRIGGER AS $$
BEGIN
    NEW.search_vector :=
        setweight(to_tsvector('english', COALESCE(NEW.original_subject, '')), 'A')
        ||
        setweight(to_tsvector('english', COALESCE(NEW.clean_text, '')), 'B');
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_processed_feedback_search_vector ON processed_feedback;
CREATE TRIGGER trg_processed_feedback_search_vector
    BEFORE INSERT OR UPDATE OF original_subject, clean_text
    ON processed_feedback
    FOR EACH ROW
    EXECUTE FUNCTION update_processed_feedback_search_vector();

-- -------------------------------------------------------
-- TRIGGER FUNCTION: auto-update updated_at on any UPDATE
-- Shared by all three tables
-- -------------------------------------------------------
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_users_updated_at ON users;
CREATE TRIGGER trg_users_updated_at
    BEFORE UPDATE ON users
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

DROP TRIGGER IF EXISTS trg_raw_feedback_updated_at ON raw_feedback;
CREATE TRIGGER trg_raw_feedback_updated_at
    BEFORE UPDATE ON raw_feedback
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

DROP TRIGGER IF EXISTS trg_processed_feedback_updated_at ON processed_feedback;
CREATE TRIGGER trg_processed_feedback_updated_at
    BEFORE UPDATE ON processed_feedback
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();
```

## 12. Full-Text Search Queries

### 12.1 Search raw_feedback by keyword (subject + description)

SELECT
    feedback_id,
